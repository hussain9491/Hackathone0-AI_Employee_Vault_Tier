## 🏗️ Implementation

### Plan Generator Script

**File:** `scripts/plan_generator.py`

```python
"""
Plan Generator for AI Employee

Creates structured action plans for complex tasks.

Usage:
    python plan_generator.py --vault-path .. --task "Setup Gmail integration"
"""

import argparse
import sys
from pathlib import Path
from datetime import datetime


class PlanGenerator:
    """Generate structured action plans."""
    
    def __init__(self, vault_path: str):
        self.vault_path = Path(vault_path)
        self.plans_folder = self.vault_path / 'Plans'
        self.plans_folder.mkdir(parents=True, exist_ok=True)
    
    def create_plan(self, topic: str, steps: list = None, 
                    priority: str = 'medium', resources: list = None) -> Path:
        """
        Create a structured action plan.
        
        Args:
            topic: Plan topic/title
            steps: List of step descriptions (optional)
            priority: low, medium, high, critical
            resources: List of resources needed
            
        Returns:
            Path to created plan file
        """
        timestamp = datetime.now()
        date_str = timestamp.strftime('%Y-%m-%d')
        
        # Create filename
        safe_topic = "".join(c if c.isalnum() else '_' for c in topic[:30])
        filename = f"Plan_{date_str}_{safe_topic}.md"
        filepath = self.plans_folder / filename
        
        # Generate steps if not provided
        if not steps:
            steps = self._generate_default_steps(topic)
        
        # Create content
        content = f'''---
created: {timestamp.isoformat()}
status: in_progress
priority: {priority}
topic: {topic}
---

# Plan: {topic}

**Created:** {date_str}  
**Status:** 🟡 In Progress  
**Priority:** {priority.capitalize()}

## Objective

{self._generate_objective(topic)}

## Steps

{self._format_steps(steps)}

## Resources Needed

{self._format_resources(resources)}

## Success Criteria

- [ ] All steps completed
- [ ] Plan moved to /Done
- [ ] Dashboard.md updated

## Notes

_Add additional notes, blockers, and observations here_

---

## Progress Log

| Date | Action | Status |
|------|--------|--------|
| {date_str} | Plan created | ✅ |

---
*Generated by Plan Generator Skill*
'''
        
        filepath.write_text(content)
        return filepath
    
    def _generate_default_steps(self, topic: str) -> list:
        """Generate default steps based on topic keywords."""
        topic_lower = topic.lower()
        
        if 'gmail' in topic_lower or 'email' in topic_lower:
            return [
                'Research Gmail API requirements',
                'Set up Google Cloud project',
                'Enable Gmail API',
                'Create OAuth credentials',
                'Install Python dependencies',
                'Create gmail_watcher.py script',
                'Test authentication flow',
                'Test email detection',
                'Add to Company_Handbook.md rules',
            ]
        elif 'whatsapp' in topic_lower:
            return [
                'Research WhatsApp Web automation',
                'Install Playwright',
                'Create whatsapp_watcher.py script',
                'Define keywords to monitor',
                'Test session persistence',
                'Test message detection',
                'Add usage guidelines to handbook',
            ]
        elif 'pdf' in topic_lower:
            return [
                'Research PDF reading libraries (PyPDF2, pdfplumber)',
                'Install dependencies',
                'Create pdf_reader.py skill',
                'Test with sample PDFs',
                'Add text extraction to action files',
                'Document in AGENT.md',
            ]
        else:
            return [
                'Define task requirements',
                'Research implementation options',
                'Create implementation plan',
                'Implement solution',
                'Test functionality',
                'Document usage',
                'Update Company_Handbook.md',
            ]
    
    def _generate_objective(self, topic: str) -> str:
        """Generate objective statement."""
        return f"Successfully implement: {topic}. This plan outlines the steps needed to complete this task and integrate it into the AI Employee system."
    
    def _format_steps(self, steps: list) -> str:
        """Format steps as markdown checklist."""
        return '\n'.join([f'- [ ] {step}' for step in steps])
    
    def _format_resources(self, resources: list = None) -> str:
        """Format resources list."""
        if not resources:
            return '- Python environment\n- Required dependencies\n- API credentials (if applicable)'
        return '\n'.join([f'- {r}' for r in resources])


def main():
    parser = argparse.ArgumentParser(description='Plan Generator for AI Employee')
    parser.add_argument('--vault-path', default='..', help='Path to Obsidian vault')
    parser.add_argument('--task', required=True, help='Task/topic to create plan for')
    parser.add_argument('--priority', default='medium', 
                        choices=['low', 'medium', 'high', 'critical'])
    parser.add_argument('--steps', nargs='+', help='Specific steps (optional)')
    
    args = parser.parse_args()
    
    print("=" * 60)
    print("📋 PLAN GENERATOR")
    print("=" * 60)
    
    generator = PlanGenerator(args.vault_path)
    filepath = generator.create_plan(
        topic=args.task,
        priority=args.priority,
        steps=args.steps
    )
    
    print(f"\n✅ Plan created: {filepath.name}")
    print(f"📁 Location: {filepath}")
    print("\nNext steps:")
    print("1. Open the plan file to review steps")
    print("2. Update Dashboard.md with new plan")
    print("3. Start working through the checklist")


if __name__ == '__main__':
    main()
```

## 🚀 Usage

### Create Plan for Task

```bash
cd AI_Employee_Vault/scripts

# Create plan for Gmail integration
python plan_generator.py --vault-path .. --task "Setup Gmail integration"

# Create plan with specific priority
python plan_generator.py --vault-path .. --task "WhatsApp monitoring" --priority high

# Create plan with custom steps
python plan_generator.py --vault-path .. --task "PDF reader" \
  --steps "Install PyPDF2" "Create reader script" "Test extraction"
```

### With Qwen

```bash
# Ask Qwen to create a plan
qwen "Create a detailed plan for implementing PDF reading functionality"

# Qwen will use plan_generator or create Plan_*.md directly
```
