## 🏗️ Implementation

### Scheduler Script

**File:** `scripts/scheduler.py`

```python
"""
Scheduler for AI Employee

Runs scheduled tasks via cron or Task Scheduler.

Usage:
    python scheduler.py --vault-path .. --task daily_briefing
    python scheduler.py --vault-path .. --task check_pending
"""

import argparse
import sys
import subprocess
from pathlib import Path
from datetime import datetime


class Scheduler:
    """Run scheduled AI Employee tasks."""
    
    def __init__(self, vault_path: str):
        self.vault_path = Path(vault_path)
        self.briefings_folder = self.vault_path / 'Briefings'
        self.briefings_folder.mkdir(parents=True, exist_ok=True)
    
    def run_daily_briefing(self) -> str:
        """
        Generate daily briefing.
        
        Returns:
            Path to briefing file
        """
        today = datetime.now()
        date_str = today.strftime('%Y-%m-%d')
        
        # Count files
        done_count = len(list((self.vault_path / 'Done').glob('*.md')))
        pending_count = len(list((self.vault_path / 'Needs_Action').glob('*.md')))
        approval_count = len(list((self.vault_path / 'Pending_Approval').glob('*.md')))
        
        # Create briefing
        briefing_content = f'''---
generated: {today.isoformat()}
type: daily_briefing
date: {date_str}
---

# Daily Briefing: {date_str}

## Executive Summary

Generated automatically by AI Employee Scheduler.

## Today's Activity

| Metric | Count |
|--------|-------|
| Completed Tasks | {done_count} |
| Pending Tasks | {pending_count} |
| Awaiting Approval | {approval_count} |

## Completed Today

{self._list_completed()}

## Pending Review

{self._list_pending()}

## Suggestions

{self._generate_suggestions(pending_count, approval_count)}

---
*Generated by Scheduler Skill*
'''
        
        # Save briefing
        filename = f"{date_str}_Daily_Briefing.md"
        filepath = self.briefings_folder / filename
        filepath.write_text(briefing_content)
        
        # Update Dashboard
        self._update_dashboard(date_str, done_count)
        
        return str(filepath)
    
    def run_weekly_audit(self) -> str:
        """
        Generate weekly business audit.
        
        Returns:
            Path to audit file
        """
        today = datetime.now()
        week_ago = today.replace(day=today.day - 7)
        
        # This would integrate with Business_Goals.md for full audit
        audit_content = f'''---
generated: {today.isoformat()}
type: weekly_audit
period: {week_ago.strftime('%Y-%m-%d')} to {today.strftime('%Y-%m-%d')}
---

# Weekly Business Audit

## Period

{week_ago.strftime('%B %d')} - {today.strftime('%B %d, %Y')}

## Revenue Summary

_Add revenue tracking here_

## Completed Projects

{self._list_completed()}

## Bottlenecks

_Identify delays and blockers_

## Cost Optimization

_Review subscriptions and expenses_

## Next Week Priorities

1. 
2. 
3. 

---
*Generated by Scheduler Skill*
'''
        
        filename = f"{today.strftime('%Y-%m-%d')}_Weekly_Audit.md"
        filepath = self.briefings_folder / filename
        filepath.write_text(audit_content)
        
        return str(filepath)
    
    def check_pending_approvals(self) -> dict:
        """Check for stale pending approvals."""
        pending_folder = self.vault_path / 'Pending_Approval'
        stale = []
        
        for f in pending_folder.glob('*.md'):
            # Check if older than 2 days
            mtime = datetime.fromtimestamp(f.stat().st_mtime)
            age = datetime.now() - mtime
            
            if age.days > 2:
                stale.append({
                    'file': f.name,
                    'age_days': age.days
                })
        
        return {'stale_count': len(stale), 'stale_files': stale}
    
    def _list_completed(self) -> str:
        """List completed files."""
        done_folder = self.vault_path / 'Done'
        files = list(done_folder.glob('*.md'))[-5:]  # Last 5
        
        if not files:
            return '_No tasks completed recently_'
        
        return '\n'.join([f"- {f.name}" for f in files])
    
    def _list_pending(self) -> str:
        """List pending files."""
        pending_folder = self.vault_path / 'Needs_Action'
        files = list(pending_folder.glob('*.md'))[:5]  # First 5
        
        if not files:
            return '_No pending tasks_'
        
        return '\n'.join([f"- {f.name}" for f in files])
    
    def _generate_suggestions(self, pending: int, approvals: int) -> str:
        """Generate suggestions based on status."""
        suggestions = []
        
        if pending > 5:
            suggestions.append("- 📋 High pending task count - consider processing batch")
        if approvals > 0:
            suggestions.append("- ⏳ Pending approvals need attention")
        if pending == 0 and approvals == 0:
            suggestions.append("- ✅ All caught up! Consider proactive improvements")
        
        return '\n'.join(suggestions) if suggestions else '_No suggestions_'
    
    def _update_dashboard(self, date_str: str, done_count: int):
        """Update Dashboard.md with briefing info."""
        dashboard_path = self.vault_path / 'Dashboard.md'
        
        if dashboard_path.exists():
            content = dashboard_path.read_text()
            
            # Add to Recent Activity
            if '## Recent Activity' in content:
                new_entry = f"- **{date_str}**: Daily briefing generated. {done_count} tasks completed."
                content = content.replace(
                    '## Recent Activity',
                    f'## Recent Activity\n{new_entry}'
                )
                dashboard_path.write_text(content)


def main():
    parser = argparse.ArgumentParser(description='Scheduler for AI Employee')
    parser.add_argument('--vault-path', default='..', help='Path to Obsidian vault')
    parser.add_argument('--task', required=True, 
                        choices=['daily_briefing', 'weekly_audit', 'check_pending'],
                        help='Task to run')
    
    args = parser.parse_args()
    
    print("=" * 60)
    print(f"⏰ SCHEDULER: {args.task.replace('_', ' ').title()}")
    print("=" * 60)
    
    scheduler = Scheduler(args.vault_path)
    
    if args.task == 'daily_briefing':
        filepath = scheduler.run_daily_briefing()
        print(f"\n✅ Daily briefing created: {filepath}")
        
    elif args.task == 'weekly_audit':
        filepath = scheduler.run_weekly_audit()
        print(f"\n✅ Weekly audit created: {filepath}")
        
    elif args.task == 'check_pending':
        result = scheduler.check_pending_approvals()
        print(f"\n📋 Stale approvals: {result['stale_count']}")
        for f in result['stale_files']:
            print(f"   - {f['file']} ({f['age_days']} days old)")


if __name__ == '__main__':
    main()
```

## 🚀 Usage

### Manual Execution

```bash
cd AI_Employee_Vault/scripts

# Run daily briefing
python scheduler.py --vault-path .. --task daily_briefing

# Run weekly audit
python scheduler.py --vault-path .. --task weekly_audit

# Check pending approvals
python scheduler.py --vault-path .. --task check_pending
```

### Schedule with cron (Linux/Mac)

**Edit crontab:**
```bash
crontab -e
```

**Add entries:**
```bash
# Daily briefing at 8:00 AM
0 8 * * * cd /path/to/AI_Employee_Vault/scripts && python scheduler.py --vault-path .. --task daily_briefing

# Weekly audit on Sunday at 9:00 AM
0 9 * * 0 cd /path/to/AI_Employee_Vault/scripts && python scheduler.py --vault-path .. --task weekly_audit

# Check pending every 6 hours
0 */6 * * * cd /path/to/AI_Employee_Vault/scripts && python scheduler.py --vault-path .. --task check_pending
```

### Schedule with Task Scheduler (Windows)

**PowerShell script:** `schedule_tasks.ps1`

```powershell
# Daily Briefing at 8 AM
$action = New-ScheduledTaskAction -Execute "python" `
  -Argument "scheduler.py --vault-path .. --task daily_briefing" `
  -WorkingDirectory "C:\Users\user1542\Documents\GitHub\H-0-Q4\AI_Employee_Vault\scripts"

$trigger = New-ScheduledTaskTrigger -Daily -At 8:00AM

Register-ScheduledTask -TaskName "AI_Employee_Daily_Briefing" `
  -Action $action -Trigger $trigger -Description "Generate daily briefing"
```

**Run:**
```powershell
.\schedule_tasks.ps1
```
